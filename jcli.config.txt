# Create group

group -a
gid sms
ok

# create user

user -a
username tester
password test
gid sms
uid tester
ok

# create smpp client connector

smppccm -a
cid smsc-test-connectoer
host smppsim
port 2776
username smppclient1
password password
src_ton 1
src_npi 1
dst_ton 1
dst_npi 1
bind ton 1
bind npi 1
bind transceiver
submit_throughput 110
ok

# start the connector

smppccm -1 smsc-test-connectoer


# configure a default MT router

mtrouter -a
type DefaultRoute
connector smppc(smsc-test-connectoer)
rate 200
ok
